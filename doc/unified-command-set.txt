Example message

message(to: 'R21') {
    serial {                                    // send the query to the serial port
        query(
            batteryStatus: true,
            version: true
        )
    }
    device(
        store: true,
        signalPower: 3,
        doppler: true
    )
    message(to: 'M') {                          // R21 will send this message to all monitors.
        serial('abcde')
    }
}
